@name DisplayPlayers
@inputs 
@outputs 
@persist T:array I C T2:array
@trigger 
if(first()|dupefinished()){
findByClass("player")
T=findToArray()
C=findCount()
#interval(1000)
I=0
while(I < C){
    print(T[I,entity]:name())
    I++
}
}
interval(100)
findByClass("player")
if(findCount()>C){
    T2=findToArray()
    print("NEW PLAYER")
    print(T2[T2:count()-1,entity]:name())
    C=findCount()
}
